}

# Multiple uses that would conflict if 'a' and 'b' weren't instantiated

    # First use: swap (Int, Str)

    result2 = swap((Bool.true, [1, 2, 3]))
    
    # Third use: swap (Str, Str)

    result3 = swap(("foo", "bar"))

}
