## Color.roc
~~~roc
## A color value.
Color := [Red, Green, Blue].{
    ## The red color.
    red : {} -> Color
    red = |{}| Red

    ## Convert a color to a string.
    to_str : Color -> Str
    to_str = |color|
        match color {
            Red => "red"
            Green => "green"
            Blue => "blue"
        }
}
~~~
