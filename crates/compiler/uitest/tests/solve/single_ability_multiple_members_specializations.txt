app [hash, hash32] { }

MHash implements
    hash : a -> U64 where a implements MHash
    hash32 : a -> U32 where a implements MHash

Id := U64 implements [MHash {hash, hash32}]

hash = \@Id n -> n
#^^^^{-1} Id#hash(4): Id -[[hash(4)]]-> U64

hash32 = \@Id n -> Num.toU32 n
#^^^^^^{-1} Id#hash32(5): Id -[[hash32(5)]]-> U32
